  const osc1 = audioCtx.createOscillator();  
  const osc2 = audioCtx.createOscillator();  
  const filter = audioCtx.createBiquadFilter();  
  const gain = audioCtx.createGain();  
  
  osc1.type = 'sawtooth';  
  osc2.type = 'square';  
  osc1.frequency.value = noteFreq;  
  osc2.frequency.value = noteFreq * 0.997;  
  
  filter.type = 'lowpass';  
  filter.frequency.value = 900;

  osc1.connect(filter);  
  osc2.connect(filter);  
  filter.connect(gain).connect(audioCtx.destination);  
  
  osc1.start(time);  
  osc2.start(time);  
}

### `Oscilloscope.jsx` — Real-Time Waveform Display

import React, { useRef, useEffect } from 'react';  
  
export default function Oscilloscope({ audioCtx, source }) {  
  const canvasRef = useRef();  
  
  useEffect(() => {  
    const canvas = canvasRef.current;  
    const ctx = canvas.getContext('2d');  
    const analyser = audioCtx.createAnalyser();  
    source.connect(analyser);  
    analyser.fftSize = 2048;  
    const bufferLength = analyser.fftSize;  
    const dataArray = new Uint8Array(bufferLength);  
  
    const draw = () => {  
      requestAnimationFrame(draw);  
      analyser.getByteTimeDomainData(dataArray);  
      ctx.fillStyle = '#111';  
      ctx.fillRect(0, 0, canvas.width, canvas.height);  
      ctx.lineWidth = 2;  
      ctx.strokeStyle = '#0f0';  
      ctx.beginPath();  
      let sliceWidth = canvas.width * 1.0 / bufferLength;  
      let x = 0;  
      for(let i=0; i<bufferLength; i++){  
        let v = dataArray[i]/128.0;  
        let y = v * canvas.height/2;  
        if(i===0){ ctx.moveTo(x,y); }  
        else { ctx.lineTo(x,y); }  
        x += sliceWidth;  
      }  
      ctx.lineTo(canvas.width, canvas.height/2);  
      ctx.stroke();  
    }  
    draw();  
  }, [audioCtx, source]);  
  
  return <canvas ref={canvasRef} width="400" height="100" className="border rounded"/>;  
}

export function createPad(audioCtx, freq, type='saw', detune=0, gainValue=0.3, time=0) {  
  const osc1 = audioCtx.createOscillator();  
  const osc2 = audioCtx.createOscillator();  
  const filter = audioCtx.createBiquadFilter();  
  const gain = audioCtx.createGain();  
  
  osc1.type = type; osc2.type = 'triangle';  
  osc1.frequency.value = freq; osc2.frequency.value = freq*(1+detune);  
  
  filter.type = 'lowpass'; filter.frequency.value = 1000; filter.Q.value = 12;  
  gain.gain.value = gainValue;  
  
  osc1.connect(filter); osc2.connect(filter);  
  filter.connect(gain).connect(audioCtx.destination);  
  
  osc1.start(time); osc2.start(time);  
  return { osc1, osc2, gain, filter };  
}

# 🎛️ Step 2 — FX Chain: Filter → Delay → Reverb → Distortion

### `src/audio/fx.js`

export function createFXChain(audioCtx, input) {  
  // Filter  
  const filter = audioCtx.createBiquadFilter();  
  filter.type = 'lowpass';  
  filter.frequency.value = 1200;  
  filter.Q.value = 8;  
  
  // Delay  
  const delay = audioCtx.createDelay();  
  delay.delayTime.value = 0.25;  
  
  // Feedback for delay  
  const feedback = audioCtx.createGain();  
  feedback.gain.value = 0.3;  
  delay.connect(feedback);  
  feedback.connect(delay);  
  
  // Reverb (convolution node)  
  const reverb = audioCtx.createConvolver();  
  // Load IR buffer for reverb externally, example: SampleManager.getSample('LargeHallIR')  
    
  // Distortion  
  const distortion = audioCtx.createWaveShaper();  
  distortion.curve = makeDistortionCurve(400); // helper function  
  distortion.oversample = '4x';  
  
  // Connect chain: input → filter → delay → reverb → distortion → destination  
  input.connect(filter);  
  filter.connect(delay);  
  delay.connect(reverb);  
  reverb.connect(distortion);  
  distortion.connect(audioCtx.destination);  
  
  return { filter, delay, feedback, reverb, distortion };  
}  
  
// Helper function to generate distortion curve  
function makeDistortionCurve(amount) {  
  const k = typeof amount === 'number' ? amount : 50;  
  const samples = 44100;  
  const curve = new Float32Array(samples);  
  const deg = Math.PI / 180;  
  for (let i = 0; i < samples; ++i) {  
    const x = (i * 2) / samples - 1;  
    curve[i] = ((3 + k) * x * 20 * deg) / (Math.PI + k * Math.abs(x));  
  }  
  return curve;  
}

### a) `Oscilloscope.jsx`

**Potential issues:**

- `source.connect(analyser)` is permanent — need **disconnect on unmount** to prevent memory leaks.
    
- `requestAnimationFrame` loop continues indefinitely — should **cancel on unmount**.
    
- Canvas size fixed — should **use `clientWidth` and `clientHeight`** for responsiveness.
    

**Fixed version:**

import React, { useRef, useEffect } from 'react';  
  
export default function Oscilloscope({ audioCtx, source }) {  
  const canvasRef = useRef();  
  const rafRef = useRef();  
  
  useEffect(() => {  
    if (!audioCtx || !source) return;  
    const canvas = canvasRef.current;  
    const ctx = canvas.getContext('2d');  
    const analyser = audioCtx.createAnalyser();  
    source.connect(analyser);  
    analyser.fftSize = 2048;  
    const bufferLength = analyser.fftSize;  
    const dataArray = new Uint8Array(bufferLength);  
  
    const draw = () => {  
      rafRef.current = requestAnimationFrame(draw);  
      analyser.getByteTimeDomainData(dataArray);  
  
      ctx.fillStyle = '#111';  
      ctx.fillRect(0, 0, canvas.width, canvas.height);  
  
      ctx.lineWidth = 2;  
      ctx.strokeStyle = '#0f0';  
      ctx.beginPath();  
  
      const sliceWidth = canvas.width / bufferLength;  
      let x = 0;  
      for (let i = 0; i < bufferLength; i++) {  
        const v = dataArray[i] / 128.0;  
        const y = v * canvas.height / 2;  
        i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);  
        x += sliceWidth;  
      }  
      ctx.lineTo(canvas.width, canvas.height / 2);  
      ctx.stroke();  
    };  
    draw();  
  
    return () => {  
      cancelAnimationFrame(rafRef.current);  
      source.disconnect(analyser);  
    };  
  }, [audioCtx, source]);  
  
  return <canvas ref={canvasRef} width="400" height="100" className="border rounded w-full h-24" />;  
}

### `WaveformVisualizer.jsx`

import React, { useRef, useEffect } from 'react';  
  
export default function WaveformVisualizer({ audioCtx, source }) {  
  const canvasRef = useRef();  
  const rafRef = useRef();  
  
  useEffect(() => {  
    if (!audioCtx || !source) return;  
    const canvas = canvasRef.current;  
    const ctx = canvas.getContext('2d');  
  
    const analyser = audioCtx.createAnalyser();  
    analyser.fftSize = 2048;  
    source.connect(analyser);  
  
    const dataArray = new Uint8Array(analyser.frequencyBinCount);  
  
    const draw = () => {  
      rafRef.current = requestAnimationFrame(draw);  
      analyser.getByteTimeDomainData(dataArray);  
  
      ctx.fillStyle = '#111';  
      ctx.fillRect(0, 0, canvas.width, canvas.height);  
  
      // Draw waveform  
      ctx.lineWidth = 2;  
      ctx.strokeStyle = '#0f0';  
      ctx.beginPath();  
      const sliceWidth = canvas.width / dataArray.length;  
      let x = 0;  
      for (let i = 0; i < dataArray.length; i++) {  
        const v = dataArray[i] / 128.0;  
        const y = v * canvas.height / 2;  
        i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);  
        x += sliceWidth;  
      }  
      ctx.stroke();  
  
      // Draw simple VU meter  
      const rms = Math.sqrt(dataArray.reduce((sum, val) => sum + Math.pow(val/128, 2), 0) / dataArray.length);  
      ctx.fillStyle = '#f00';  
      ctx.fillRect(canvas.width - 10, canvas.height - rms*canvas.height, 8, rms*canvas.height);  
    };  
  
    draw();  
  
    return () => cancelAnimationFrame(rafRef.current);  
  }, [audioCtx, source]);  
  
  return <canvas ref={canvasRef} width="400" height="80" className="border rounded w-full h-20" />;  
}

## 2️⃣ `fx.js` — Safe FX Chain

export function createFXChain(audioCtx, input, options={}) {  
  const { filterFreq=1200, delayTime=0.25, feedbackGain=0.25, distortionAmount=300, reverbBuffer=null } = options;  
  
  const filter = audioCtx.createBiquadFilter();  
  filter.type='lowpass'; filter.frequency.value = filterFreq; filter.Q.value = 8;  
  
  const delay = audioCtx.createDelay();  
  delay.delayTime.value = delayTime;  
  const feedback = audioCtx.createGain();  
  feedback.gain.value = Math.min(feedbackGain,0.5);  
  delay.connect(feedback); feedback.connect(delay);  
  
  const distortion = audioCtx.createWaveShaper();  
  distortion.curve = makeDistortionCurve(distortionAmount);  
  distortion.oversample='4x';  
  
  const reverb = audioCtx.createConvolver();  
  if(reverbBuffer) reverb.buffer = reverbBuffer;  
  
  input.connect(filter); filter.connect(delay); delay.connect(reverb); reverb.connect(distortion); distortion.connect(audioCtx.destination);  
  
  return { filter, delay, feedback, reverb, distortion };  
}  
  
function makeDistortionCurve(amount){ const k=amount||50; const samples=44100; const curve=new Float32Array(samples); const deg=Math.PI/180;  
  for(let i=0;i<samples;i++){ const x=(i*2)/samples-1; curve[i]=((3+k)*x*20*deg)/(Math.PI+k*Math.abs(x)); } return curve;  
}

// SpectrogramVisualizer.jsx  
import React, { useRef, useEffect } from 'react';  
  
export default function SpectrogramVisualizer({ audioCtx, source }) {  
  const canvasRef = useRef();  
  const rafRef = useRef();  
  
  useEffect(() => {  
    if(!audioCtx || !source) return;  
    const canvas = canvasRef.current;  
    const ctx = canvas.getContext('2d');  
    const analyser = audioCtx.createAnalyser();  
    analyser.fftSize = 2048;  
    source.connect(analyser);  
    const dataArray = new Uint8Array(analyser.frequencyBinCount);  
  
    const draw = () => {  
      rafRef.current = requestAnimationFrame(draw);  
      analyser.getByteFrequencyData(dataArray);  
      ctx.fillStyle = '#111';  
      ctx.fillRect(0,0,canvas.width,canvas.height);  
  
      const barWidth = canvas.width/dataArray.length;  
      dataArray.forEach((val,i)=>{  
        ctx.fillStyle = `rgb(${val},${255-val},${val/2})`;  
        ctx.fillRect(i*barWidth,canvas.height-val/2,barWidth,val/2);  
      });  
    };  
  
    draw();  
    return () => cancelAnimationFrame(rafRef.current);  
  }, [audioCtx, source]);  
  
  return <canvas ref={canvasRef} width="600" height="150" className="border rounded w-full" />;  
}

export class PolySynth {  
  constructor(audioCtx, waveform='sawtooth') {  
    this.audioCtx = audioCtx;  
    this.voices = [];  
    this.waveform = waveform;  
  }  
  
  trigger(note, duration=0.5, volume=0.5) {  
    const osc = this.audioCtx.createOscillator();  
    const gain = this.audioCtx.createGain();  
    osc.type = this.waveform;  
    osc.frequency.value = 440 * Math.pow(2, (note-69)/12); // MIDI note  
    gain.gain.value = volume;  
    osc.connect(gain); gain.connect(this.audioCtx.destination);  
    osc.start();  
    osc.stop(this.audioCtx.currentTime + duration);  
    this.voices.push({osc,gain});  
  }  
}

// Analyze track to generate sequencer steps  
export function generateSequencerPattern(track, album='Discovery') {  
  const bpm = track.tempo || 120;  
  const energy = track.energy || 0.5;  
  
  // Map to step sequencer  
  const steps = 16;  
  const tracks = [...AlbumPatches[album].drums, ...AlbumPatches[album].synths];  
  
  const pattern = tracks.map(t=>{  
    const row = Array(steps).fill(false);  
    for(let i=0;i<steps;i++){  
      if(Math.random() < energy) row[i] = true; // higher energy → more active steps  
    }  
    return { track: t, steps: row };  
  });  
  return { bpm, pattern };  
}

import * as THREE from 'three';  
  
export class AudioParticles {  
  constructor(audioCtx, analyser, container) {  
    this.audioCtx = audioCtx;  
    this.analyser = analyser;  
    this.container = container;  
    this.scene = new THREE.Scene();  
    this.camera = new THREE.PerspectiveCamera(75, container.clientWidth/container.clientHeight, 0.1, 1000);  
    this.renderer = new THREE.WebGLRenderer({ antialias: true });  
    this.renderer.setSize(container.clientWidth, container.clientHeight);  
    container.appendChild(this.renderer.domElement);  
  
    const geometry = new THREE.BufferGeometry();  
    const numParticles = 1024;  
    const positions = new Float32Array(numParticles * 3);  
    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));  
      
    const material = new THREE.PointsMaterial({ color: 0x00ff00, size: 0.05 });  
    this.particles = new THREE.Points(geometry, material);  
    this.scene.add(this.particles);  
  
    this.camera.position.z = 5;  
  
    this.dataArray = new Uint8Array(analyser.frequencyBinCount);  
    this.animate();  
  }  
  
  animate = () => {  
    requestAnimationFrame(this.animate);  
    this.analyser.getByteFrequencyData(this.dataArray);  
    const positions = this.particles.geometry.attributes.position.array;  
    for (let i = 0; i < this.dataArray.length; i++) {  
      positions[i*3] = Math.sin(i*0.1 + performance.now()*0.001) * this.dataArray[i]/128;  
      positions[i*3+1] = Math.cos(i*0.1) * this.dataArray[i]/128;  
      positions[i*3+2] = Math.sin(i*0.2) * this.dataArray[i]/128;  
    }  
    this.particles.geometry.attributes.position.needsUpdate = true;  
    this.renderer.render(this.scene, this.camera);  
  }  
}

export function createFXChain(audioCtx, input, options = {}) {  
  const { filterFreq = 1200, delayTime = 0.25, feedbackGain = 0.25, distortionAmount = 300, reverbBuffer = null } = options;  
  
  const filter = audioCtx.createBiquadFilter();  
  filter.type = 'lowpass';  
  filter.frequency.value = filterFreq;  
  filter.Q.value = 8;  
  
  const delay = audioCtx.createDelay();  
  delay.delayTime.value = delayTime;  
  
  const fbGain = audioCtx.createGain();  
  fbGain.gain.value = Math.min(feedbackGain, 0.5);  
  delay.connect(fbGain);  
  fbGain.connect(delay);  
  
  const distortion = audioCtx.createWaveShaper();  
  distortion.curve = makeDistortionCurve(distortionAmount);  
  distortion.oversample = '4x';  
  
  const reverb = audioCtx.createConvolver();  
  if (reverbBuffer) reverb.buffer = reverbBuffer;  
  
  input.connect(filter);  
  filter.connect(delay);  
  delay.connect(reverb);  
  reverb.connect(distortion);  
  distortion.connect(audioCtx.destination);  
  
  return { filter, delay, fbGain, reverb, distortion };  
}  
  
function makeDistortionCurve(amount) {  
  const k = amount || 50;  
  const samples = 44100;  
  const curve = new Float32Array(samples);  
  const deg = Math.PI / 180;  
  for (let i = 0; i < samples; i++) {  
    const x = (i * 2) / samples - 1;  
    curve[i] = ((3 + k) * x * 20 * deg) / (Math.PI + k * Math.abs(x));  
  }  
  return curve;  
}

export class PolySynth {  
  constructor(audioCtx, waveform = 'sawtooth') {  
    this.audioCtx = audioCtx;  
    this.voices = [];  
    this.waveform = waveform;  
  }  
  
  trigger(note, duration = 0.5, volume = 0.5) {  
    if (!this.audioCtx) return;  
    const osc = this.audioCtx.createOscillator();  
    const gain = this.audioCtx.createGain();  
    osc.type = this.waveform;  
    osc.frequency.value = 440 * Math.pow(2, (note - 69) / 12);  
    gain.gain.value = volume;  
    osc.connect(gain);  
    gain.connect(this.audioCtx.destination);  
    osc.start();  
    osc.stop(this.audioCtx.currentTime + duration);  
    this.voices.push({ osc, gain });  
  }  
}

      const positions = new Float32Array(1024 * 3);  
      geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));  
      const material = new THREE.PointsMaterial({ color: new THREE.Color(`hsl(${idx * 60}, 100%, 50%)`), size: 0.08 });  
      const points = new THREE.Points(geometry, material);  
      scene.add(points);  
      return { analyser, points, positions };  
    });  
  
    const dataArray = new Uint8Array(512);  
  
    const animate = () => {  
      rafRef.current = requestAnimationFrame(animate);  
      particleSystems.forEach((ps, i) => {  
        ps.analyser?.getByteFrequencyData(dataArray);  
        for (let j = 0; j < dataArray.length; j++) {  
          const v = dataArray[j] / 128;  
          ps.positions[j * 3] = Math.sin(j * 0.1 + performance.now() * 0.001) * v;  
          ps.positions[j * 3 + 1] = Math.cos(j * 0.1) * v;  
          ps.positions[j * 3 + 2] = Math.sin(j * 0.2) * v;  
        }  
        ps.points.geometry.attributes.position.needsUpdate = true;  
        if (fxValues[i]) {  
          ps.points.material.size = 0.05 + (fxValues[i].filter || 0) / 500;  
          ps.points.material.color.setHSL((fxValues[i].filter || 0) / 1000, 1, 0.5);  
        }  
      });  
      renderer.render(scene, camera);  
    };  
  
    animate();  
    return () => cancelAnimationFrame(rafRef.current);  
  }, [analysers, fxValues]);  
  
  return <div ref={containerRef} className="w-full h-96 border rounded" />;  
}
