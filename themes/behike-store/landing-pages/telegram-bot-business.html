async def get_budget(update: Update, context: ContextTypes.DEFAULT_TYPE):
    budget = update.message.text
    context.user_data["budget"] = budget

    # Score the lead based on answers
    score = calculate_score(context.user_data)

    if score >= 70:
        # Strong fit — send calendar link immediately
        await update.message.reply_text(
            "You're a strong fit. Book a call here: calendly.com/yourlink"
        )
        await send_notification(YOUR_CHAT_ID, f"HOT LEAD (score: {score})")

    elif score >= 40:
        # Warm lead — manual follow-up
        await update.message.reply_text(
            "We'll review your answers and reach out within 48 hours."
        )

    else:
        # Not a fit — redirect to free resources
        await update.message.reply_text(
            "Check out our free resources at behike.com/free."
        )

    return ConversationHandler.END
